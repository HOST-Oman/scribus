		add_library(${SCRIBUS_IMPORTPM_PLUGIN} MODULE ${IMPORTPM_PLUGIN_SOURCES} ${IMPORTPM_PLUGIN_MOC_SOURCES})

		add_dependencies(${SCRIBUS_IMPORTPM_PLUGIN} ${EXE_NAME})
