include_directories(
	${CMAKE_SOURCE_DIR}
	${CMAKE_SOURCE_DIR}/scribus
)

	importxar.h
	importxarplugin.h
)

#	qtiocompressor.cpp
	importxar.cpp
	importxarplugin.cpp
)
