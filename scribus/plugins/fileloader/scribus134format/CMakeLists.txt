include_directories(
	${CMAKE_SOURCE_DIR}
	${CMAKE_SOURCE_DIR}/scribus
)

	scribus134format.h
	scribus134formatimpl.h
)

	scribus134format.cpp
	scribus134formatimpl.cpp
)

set(SCRIBUS_SCR134FORMAT_FL_PLUGIN "scribus134format")

	LIBRARY
	DESTINATION ${PLUGINDIR}
